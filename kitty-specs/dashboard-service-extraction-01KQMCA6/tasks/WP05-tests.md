## Context

The architectural boundary rule (FR-010) is intentionally narrower than the
full "no specify_cli imports" goal:

> No module inside `src/dashboard/` imports from `src/specify_cli/dashboard/`

This prevents the circular dependency between the service layer and its own
adapter. `src/dashboard/` may still import from `specify_cli.scanner`,
`specify_cli.status.*`, `specify_cli.sync.daemon`, etc. — those are legitimate
cross-layer dependencies on subsystems not yet extracted.

The seam tests follow DIRECTIVE_036 (black-box integration testing):
- The **CLI adapter** is exercised for real (its handler method is called)
- **External I/O** (scanner, filesystem, daemon) is stubbed with fixtures
- Assertions are on **observable response output** (the dict returned by the
  service layer, serialized to JSON by the adapter)

Read `tests/test_events_tracker_public_imports.py` for the AST-scan pattern
used in other boundary tests. Read `tests/test_dashboard/` for the existing
fixture and test patterns.

   ```python
   """Boundary test: src/dashboard/ must not import from src/specify_cli/dashboard/.

   This prevents the circular dependency between the service layer and its
   own CLI adapter. See research.md §4 for the staged-boundary rationale.
   """
   from __future__ import annotations

   import ast
   from pathlib import Path

   import pytest

   DASHBOARD_SRC = Path(__file__).parent.parent.parent / "src" / "dashboard"
   FORBIDDEN_PREFIX = "specify_cli.dashboard"


   def _collect_imports(path: Path) -> list[str]:
       """Return all import module names found in a Python source file."""
       try:
           tree = ast.parse(path.read_text(encoding="utf-8"))
       except SyntaxError:
           return []
       names: list[str] = []
       for node in ast.walk(tree):
           if isinstance(node, ast.Import):
               names.extend(alias.name for alias in node.names)
           elif isinstance(node, ast.ImportFrom):
               if node.module:
                   names.append(node.module)
       return names


   def test_dashboard_has_no_imports_from_specify_cli_dashboard() -> None:
       """src/dashboard/ must not import from src/specify_cli/dashboard/."""
       violations: list[str] = []
       for py_file in DASHBOARD_SRC.rglob("*.py"):
           for import_name in _collect_imports(py_file):
               if import_name.startswith(FORBIDDEN_PREFIX):
                   violations.append(f"{py_file.relative_to(DASHBOARD_SRC.parent.parent)}: {import_name}")
       assert not violations, (
           "src/dashboard/ imports from src/specify_cli/dashboard/ (circular dependency):\n"
           + "\n".join(f"  - {v}" for v in violations)
       )
   ```


   ```python
   """Seam delegation tests: CLI adapter → service layer.

   Each test verifies that the handler method:
     1. Delegates to the expected service object
     2. Returns the service response as JSON

   Following DIRECTIVE_036: tests exercise the adapter through its HTTP
   interface; external I/O is stubbed.
   """
   from __future__ import annotations
   from unittest.mock import MagicMock, patch
   import json
   import pytest


   class TestMissionScanServiceSeams:

       def test_features_list_delegates_to_mission_scan_service(self, mock_handler):
           """handle_features_list must delegate to MissionScanService.get_features_list."""
           expected = {"features": [], "active_feature_id": None, ...}  # minimal FeaturesListResponse
           with patch("specify_cli.dashboard.handlers.features.MissionScanService") as MockService:
               MockService.return_value.get_features_list.return_value = expected
               mock_handler.handle_features_list()
               MockService.assert_called_once()
               MockService.return_value.get_features_list.assert_called_once()
               # Assert response was serialized to JSON
               written = b"".join(mock_handler.wfile_written)
               assert json.loads(written) == expected

       def test_kanban_delegates_to_mission_scan_service(self, mock_handler):
           """handle_kanban must delegate to MissionScanService.get_kanban."""
           expected = {"lanes": {}, "is_legacy": False, "upgrade_needed": False, "weighted_percentage": None}
           with patch("specify_cli.dashboard.handlers.features.MissionScanService") as MockService:
               MockService.return_value.get_kanban.return_value = expected
               mock_handler.handle_kanban("/api/kanban/my-feature-01KQMCA6")
               MockService.return_value.get_kanban.assert_called_once_with("my-feature-01KQMCA6")
               written = b"".join(mock_handler.wfile_written)
               assert json.loads(written) == expected
   ```
